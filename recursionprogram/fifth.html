#include<stdio.h>  
  
            #define N 5  
              
            void smallest(int *num, int n, int small)  
            {  
                if(n < 0)  
                    printf("Smallest Element In The Array: %d\n", small);  
                else  
                {  
                    if(small > *num)  
                        small = *num;  
              
                    smallest(++num, --n, small);  
                }  
            }  
              
            int main()  
            {  
                int a[N], i;  
              
                printf("Enter %d integer numbers\n", N);  
                for(i = 0; i < N; i++)  
                    scanf("%d", &a[i]);  
              
                smallest(a, N - 1, a[0]);  
              
                return 0;  
            }
            OUTPUT:
            Enter 5 integer numbers
2 3  56 7 8 
Smallest Element In The Array: 2
