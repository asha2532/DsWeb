#include<stdio.h>
            int comparestrings(char *str1,char *str2)
            {
                int i=0;
                while(str1[i]==str2[i])
                {
                    if(str1[i]=='\0')
                    {
                        return 0;
                    }
                    i++;
                }
                return str1[i]-str2[i];
            }
            int main()
            {
                char str1[]="haha";
                 char str2[]="haha";
                 int result=comparestrings(str1,str2);
                 if(result==0)
                 {
                     printf("the strings are equal \n");
                     
                 }
                 else
                 {
                     printf("not equal");
                 }
                 
                 return 0;
            }
            OUTPUT:
            the strings are equal
