#include<stdio.h>
            int stringlength(char*str)
            {
                int length=0;
                while(str[length]!='\0')
                {
                    length++;
                }
                return length;
            }
            int main()
            {
                char str[]="Hello,world!";
                int length=stringlength(str);
                printf("the final length of the string is :%d\n",length);
            
                return 0;
            }
            OUTPUT:
            the final length of the string is :12
