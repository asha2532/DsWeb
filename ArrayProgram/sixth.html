#include <stdio.h>

            int main()
            {
                int arr[5];
                int i=0;
                
            
                printf("Array elements: \n");
                for (i = 0; i < 5; i++)
                    scanf("%d ",&arr[i]);
            
                printf("\nSquare of array elements: \n");
                for (i = 0; i < 5; i++)
                    printf("%d ", arr[i] * arr[i]);
            
                printf("\n");
            
                return 0;
            }
            OUTPUT:
            Array elements:
             1 2 3 4 
            Square of array elements: 
              1 4 9 16
